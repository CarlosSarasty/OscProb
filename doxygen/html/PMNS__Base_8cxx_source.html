717 void PMNS_Base::RotateH(int i,int j){
718 
719  // Do nothing if angle is zero
720  if(fTheta[i][j]==0) return;
721 
722  double fSinBuffer = sin(fTheta[i][j]);
723  double fCosBuffer = cos(fTheta[i][j]);
724 
725  double fHmsBufferD;
726  complex fHmsBufferC;
727 
728  // With Delta
729  if(i+1<j){
730  complex fExpBuffer = complex(cos(fDelta[i][j]), -sin(fDelta[i][j]));
731 
732  // General case
733  if(i>0){
734  // Top columns
735  for(int k=0; k<i; k++){
736  fHmsBufferC = fHms[k][i];
737 
738  fHms[k][i] *= fCosBuffer;
739  fHms[k][i] += fHms[k][j] * fSinBuffer * conj(fExpBuffer);
740 
741  fHms[k][j] *= fCosBuffer;
742  fHms[k][j] -= fHmsBufferC * fSinBuffer * fExpBuffer;
743  }
744 
745  // Middle row and column
746  for(int k=i+1; k<j; k++){
747  fHmsBufferC = fHms[k][j];
748 
749  fHms[k][j] *= fCosBuffer;
750  fHms[k][j] -= conj(fHms[i][k]) * fSinBuffer * fExpBuffer;
751 
752  fHms[i][k] *= fCosBuffer;
753  fHms[i][k] += fSinBuffer * fExpBuffer * conj(fHmsBufferC);
754  }
755 
756  // Nodes ij
757  fHmsBufferC = fHms[i][i];
758  fHmsBufferD = real(fHms[j][j]);
759 
760  fHms[i][i] *= fCosBuffer * fCosBuffer;
761  fHms[i][i] += 2 * fSinBuffer * fCosBuffer * real(fHms[i][j] * conj(fExpBuffer));
762  fHms[i][i] += fSinBuffer * fHms[j][j] * fSinBuffer;
763 
764  fHms[j][j] *= fCosBuffer * fCosBuffer;
765  fHms[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
766  fHms[j][j] -= 2 * fSinBuffer * fCosBuffer * real(fHms[i][j] * conj(fExpBuffer));
767 
768  fHms[i][j] -= 2 * fSinBuffer * real(fHms[i][j] * conj(fExpBuffer)) * fSinBuffer * fExpBuffer;
769  fHms[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC) * fExpBuffer;
770 
771  }
772  // First rotation on j (No top columns)
773  else{
774  // Middle rows and columns
775  for(int k=i+1; k<j; k++){
776  fHms[k][j] = -conj(fHms[i][k]) * fSinBuffer * fExpBuffer;
777 
778  fHms[i][k] *= fCosBuffer;
779  }
780 
781  // Nodes ij
782  fHmsBufferD = real(fHms[i][i]);
783 
784  fHms[i][j] = fSinBuffer * fCosBuffer * (fHms[j][j] - fHmsBufferD) * fExpBuffer;
785 
786  fHms[i][i] *= fCosBuffer * fCosBuffer;
787  fHms[i][i] += fSinBuffer * fHms[j][j] * fSinBuffer;
788 
789  fHms[j][j] *= fCosBuffer * fCosBuffer;
790  fHms[j][j] += fSinBuffer * fHmsBufferD * fSinBuffer;
791  }
792 
793  }
794  // Without Delta (No middle rows or columns: j = i+1)
795  else{
796  // General case
797  if(i>0){
798  // Top columns
799  for(int k=0; k<i; k++){
800  fHmsBufferC = fHms[k][i];
801 
802  fHms[k][i] *= fCosBuffer;
803  fHms[k][i] += fHms[k][j] * fSinBuffer;
804 
805  fHms[k][j] *= fCosBuffer;
806  fHms[k][j] -= fHmsBufferC * fSinBuffer;
807  }
808 
809  // Nodes ij
810  fHmsBufferC = fHms[i][i];
811  fHmsBufferD = real(fHms[j][j]);
812 
813  fHms[i][i] *= fCosBuffer * fCosBuffer;
814  fHms[i][i] += 2 * fSinBuffer * fCosBuffer * real(fHms[i][j]);
815  fHms[i][i] += fSinBuffer * fHms[j][j] * fSinBuffer;
816 
817  fHms[j][j] *= fCosBuffer * fCosBuffer;
818  fHms[j][j] += fSinBuffer * fHmsBufferC * fSinBuffer;
819  fHms[j][j] -= 2 * fSinBuffer * fCosBuffer * real(fHms[i][j]);
820 
821  fHms[i][j] -= 2 * fSinBuffer * real(fHms[i][j]) * fSinBuffer;
822  fHms[i][j] += fSinBuffer * fCosBuffer * (fHmsBufferD - fHmsBufferC);
823 
824  }
825  // First rotation (theta12)
826  else{
827 
828  fHms[i][j] = fSinBuffer * fCosBuffer * fHms[j][j];
829 
830  fHms[i][i] = fSinBuffer * fHms[j][j] * fSinBuffer;
831 
832  fHms[j][j] *= fCosBuffer * fCosBuffer;
833 
834  }
835  }

884 {
885 
886  // Set the neutrino path
887  SetCurPath(p);
888 
889  // Solve for eigensystem
890  SolveHam();

1076 double PMNS_Base::AvgProb(int flvi, int flvf, double E, double dE)
1077 {
1078 
1079  // Do nothing if energy is not positive
1080  if(E<=0) return 0;
1081 
1082  if(fNuPaths.empty()) return 0;
1083 
1084  if(fNuPaths.size() != 1){
1085  cout << "ERROR: AvgProb not implemented for multiple paths." << endl;
1086  cout << " Returning probability at bin center." << endl;
1087  return Prob(flvi, flvf, E);
1088  }
1089 
1090  // Don't average zero width
1091  if(dE<=0) return Prob(flvi, flvf, E);
1092 
1093  // Make sure fPath is set
1094  SetCurPath(fNuPaths[0]);
1095 
1096  // Define L/E variables
1097  double LoE = 0;
1098  double dLoE = 0;
1099 
1100  // Set a minimum energy
1101  double minE = 0.1 * E;
1102 
1103  // Transform range to L/E
1104  // Full range if low edge > minE
1105  if(E-dE/2 > minE){
1106  LoE = 0.5 * (fPath.length/(E-dE/2) + fPath.length/(E+dE/2));
1107  dLoE = fPath.length/(E-dE/2) - fPath.length/(E+dE/2);
1108  }

1215 vector<double> PMNS_Base::GetSamplePoints(double LoE, double dLoE)
1216 {
1217 
1218  // Solve Hamiltonian to get eigenvalues
1219  SolveHam();
1220 
1221  // Define conversion factor [km/GeV -> 1/(4 eV^2)]
1222  const double k1267 = kKm2eV / (4 * kGeV2eV);
1223 
1224  // Get list of all effective Dm^2
1225  vector<double> effDm;
1226 
1227  for(int i=0; i<fNumNus-1; i++){
1228  for(int j=i+1; j<fNumNus; j++){
1229  effDm.push_back( 2 * kGeV2eV * fEnergy * fabs(fEval[j] - fEval[i]) );
1230  }
1231  }
1232 
1233  int numDm = effDm.size();
1234 
1235  // Set a number of sub-divisions to achieve "good" accuracy
1236  // This needs to be studied better
1237  int n_div = ceil( 20 * pow(dLoE/LoE,0.8) );
1238  //int n_div = 1;
1239

1241  vector<double> allSamples;
1242 
1243  // Loop over sub-divisions
1244  for(int k=0; k<n_div; k++){
1245

1247  double bctr = LoE - dLoE/2 + (k+0.5)*dLoE/n_div;
1248  double bwdt = dLoE/n_div;
1249 
1250  // Make a list of indices sorted by Dm^2 value
1251  vector<int> Idx(numDm, 0);
1252  for(int i=0; i<numDm; i++) Idx[i] = i;
1253  sort(Idx.begin(), Idx.end(), IdxCompare(effDm));
1254 
1255  // Make a vector of L/E sample values
1256  // Initialized in the sub-division center
1257  vector<double> samples;
1258  samples.push_back(bctr);
1259 
1260  // Loop over all Dm^2 to average each frequency
1261  // This will recursively sample points in smaller
1262  // bins so that all relevant frequencies are used
1263  for(int i=0; i<numDm; i++){
1264 
1265  // Copy the list of sample L/E values
1266  vector<double> prev = samples;
1267 
1268  // Redefine bin width to lie within full sub-division
1269  double Width = 2*min(prev[0] - (bctr - bwdt/2), (bctr + bwdt/2) - prev[0]);
1270 
1271  // Compute oscillation argument sorted from lowest to highest
1272  const double arg = k1267 * effDm[Idx[i]] * Width;
1273 
1274  // Skip small oscillation values.
1275  // If it's the last one, lower the tolerance
1276  if(i < numDm-1){
1277  if(arg<0.9) continue;
1278  }
1279  else{
1280  if(arg<0.1) continue;
1281  }
1282 
1283  // Reset samples to redefine them
1284  samples.clear();
1285 
1286  // Loop over previous samples
1287  for(int j=0; j<int(prev.size()); j++){
1288 
1289  // Compute new sample points around old samples
1290  // This is based on a oscillatory quadrature rule
1291  double sample = (1/sqrt(3)) * (Width/2);
1292  if(arg!=0) sample = acos(sin(arg)/arg)/arg * (Width/2);
1293 
1294  // Add samples above and below center
1295  samples.push_back(prev[j]-sample);
1296  samples.push_back(prev[j]+sample);
1297 
1298  }

1300  }// End of loop over Dm^2
1301

1303  allSamples.insert(allSamples.end(), samples.begin(), samples.end());
