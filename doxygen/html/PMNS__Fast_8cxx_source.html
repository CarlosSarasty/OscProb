76 void PMNS_Fast::UpdateHam()
77 {
78 
79  double lv = 2 * kGeV2eV*fEnergy; // 2E in eV
80 
81  double kr2GNe = kK2*M_SQRT2*kGf;
82  kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
83 
84  // Finish building Hamiltonian in matter with dimension of eV
85  for(int i=0;i<fNumNus;i++){
86  fHam[i][i] = fHms[i][i]/lv;

88  if(!fIsNuBar) fHam[i][j] = fHms[i][j]/lv;

93  else fHam[0][0] -= kr2GNe;

146 void PMNS_Fast::SetVacuumEigensystem()
147 {
148 
149  double s12, s23, s13, c12, c23, c13;
150  complex idelta(0.0, fDelta[0][2]);
151  if(fIsNuBar) idelta = conj(idelta);
152 
153  s12 = sin(fTheta[0][1]); s23 = sin(fTheta[1][2]); s13 = sin(fTheta[0][2]);
154  c12 = cos(fTheta[0][1]); c23 = cos(fTheta[1][2]); c13 = cos(fTheta[0][2]);
155 
156  fEvec[0][0] = c12*c13;
157  fEvec[0][1] = s12*c13;
158  fEvec[0][2] = s13*exp(-idelta);
159 
160  fEvec[1][0] = -s12*c23-c12*s23*s13*exp(idelta);
161  fEvec[1][1] = c12*c23-s12*s23*s13*exp(idelta);
162  fEvec[1][2] = s23*c13;
163 
164  fEvec[2][0] = s12*s23-c12*c23*s13*exp(idelta);
165  fEvec[2][1] = -c12*s23-s12*c23*s13*exp(idelta);
166  fEvec[2][2] = c23*c13;
167 
168  fEval[0] = 0;
169  fEval[1] = fDm[1] / (2 * kGeV2eV*fEnergy);
